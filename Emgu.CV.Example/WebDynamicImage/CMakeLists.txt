PROJECT(Example.WebDynamicImage)

GET_TARGET_PROPERTY(CXCORE_LOCATION cxcore LOCATION)
GET_TARGET_PROPERTY(CV_LOCATION cv LOCATION)
GET_TARGET_PROPERTY(HIGHGUI_LOCATION highgui LOCATION)
GET_TARGET_PROPERTY(CVAUX_LOCATION cvaux LOCATION)
GET_TARGET_PROPERTY(CVEXTERN_LOCATION cvextern LOCATION)
GET_TARGET_PROPERTY(ML_LOCATION ml LOCATION)

ADD_CUSTOM_TARGET(
   ${PROJECT_NAME} ALL
   COMMAND ${CMAKE_COMMAND} -E copy ${CXCORE_LOCATION} ${CMAKE_CURRENT_SOURCE_DIR}/Bin
   COMMAND ${CMAKE_COMMAND} -E copy ${CV_LOCATION} ${CMAKE_CURRENT_SOURCE_DIR}/Bin
   COMMAND ${CMAKE_COMMAND} -E copy ${CVAUX_LOCATION} ${CMAKE_CURRENT_SOURCE_DIR}/Bin
   COMMAND ${CMAKE_COMMAND} -E copy ${HIGHGUI_LOCATION} ${CMAKE_CURRENT_SOURCE_DIR}/Bin
   COMMAND ${CMAKE_COMMAND} -E copy ${ML_LOCATION} ${CMAKE_CURRENT_SOURCE_DIR}/Bin
)
ADD_DEPENDENCIES(${PROJECT_NAME} Emgu.CV Emgu.CV.UI cv cxcore cvaux highgui cvextern)
